sample text file for adding git repo from local repo

git clone https://github.com/katha60km/Test.git

git config --global user.name katha60km

git config --global user.email ramukumarvsmi@gmail.com

git add <FileName>
    # it will added to staging area

git commit <FileName>
    # it will ready for push

git push origin main
    # File was sucessfully upload to github

git pull 
    # download modified file from github

git init
    # when we created a new folder from local repo we use "git init"

git remote add orgin <new git repo extension>
    # create a new folder under test and connect to local machine
